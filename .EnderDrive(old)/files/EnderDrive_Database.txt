Table folder {
  id integer [primary key]
  name varchar
  size integer
  created_at timestamp
  parent_id integer
  owner_id integer
  view_access integer
  full_access integer
  files_id integer
}

Table users {
  id integer [primary key]
  username varchar
  password varchar
  role varchar
  created_at timestamp
}

Table file {
  id integer [primary key]
  name varchar
  size integer
  created_at timestamp
  owner_id integer
  view_access integer
  full_access integer
}

Ref: file.id > folder.files_id
Ref: file.owner_id > users.id
Ref: file.view_access <> users.id
Ref: file.full_access <> users.id

Ref: folder.owner_id - users.id
Ref: folder.view_access <> users.id
Ref: folder.full_access <> users.id
